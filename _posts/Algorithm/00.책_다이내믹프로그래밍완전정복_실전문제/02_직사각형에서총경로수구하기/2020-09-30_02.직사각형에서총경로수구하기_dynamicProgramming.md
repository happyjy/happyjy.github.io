```js
function numOfPaths(m, n) {
  var cache = Array(m).map(v => Array(n))

  for (var i = 0; i < m; i++) {
    cache[i][0] = 1
  }
  for (var j = 0; j < n; j++) {
    cache[0][j] = 1
  }

  //# step1
  // 나머지 셀
  for (var i = 1; i < m; i++) {
    for (var i = 1; i < m; i++) {
      cache[i][j] = cache[i - 1][j] + cache[i][j - 1] // cache[i][j]기준 위 셀 + cache[i][j]기준 왼쪽 셀
    }
    //# step2,3
  }

  return cache[m - 1][n - 1]
}

console.log(numOfPaths(2, 3))
```
