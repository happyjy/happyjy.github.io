## 8. 철근 자르기
### 하향식 접근
maxValue(N) = MAX(
                value[1] + maxValue(N-1),
                value[2] + maxValue(N-2),
                
                value[N] + maxValue(N-N)
                )

maxValue(1) = MAX(value[1] + maxValue(0))

maxValue(2) = MAX(
                  value[1] + maxValue(1)
                  value[2] + maxValue(0)
                  )

maxValue(3) = MAX(
                  value[1] + maxValue(2)
                  value[2] + maxValue(1)
                  value[3] + maxValue(0)
                  ) 

maxValue(4) = MAX(
                  value[1] + maxValue(3)
                  value[2] + maxValue(2)
                  value[3] + maxValue(1)
                  value[4] + maxValue(0)
                  )
### 상향식 접근
for(var i = 1; i <= N; i++){
  maxValues[i] = -1; //INT_MIN;
  for(var j = 1; j <=i; j++){
    maxValues[i] = Math.min(maxValues[i], value[j] + maxValues[i-j])
  }
}
